In [7]:
class IterClass(object):
    
    
    def __init__(self, start, last):
        self.current = start
        self.max = last
        
    #  'object is not iterable' Exception 
    def __iter__(self):       
        return self
    
    def __next__(self):
        if self.current > self.max:
            raise StopIteration
        else:
            self.current += 1
            return self.current-1
