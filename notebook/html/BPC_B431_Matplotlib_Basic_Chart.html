# 지연 랜더링, deferred rendering 메커니즘은 맷플롯립의 핵심

import numpy as np
import matplotlib.pyplot as plt

def plot_slope(X, Y):
    Xs = X[1:] - X[:-1]
    Ys = Y[1:] - Y[:-1]
    plt.plot(X[1:], Ys/Xs)
    
X = np.linspace(-3, 3, 100)
Y = np.exp(-X**2)

plt.plot(X, Y)
plot_slope(X, Y)
plt.show()

# 반복자 enumerate 는 현재 행과 해당 첨자를 둘 다 반환
import numpy as np
import matplotlib.pyplot as plt

data = [[5., 25., 50., 20.],
        [4., 23., 51., 17.],
        [6., 22., 52., 19.]]

color_list = ['b', 'g', 'r']
gap = .8 / len(data)
for i, row in enumerate(data):
    X = np.arange(len(row))
    plt.bar(X + i*gap, row, 
           width = gap, 
           color = color_list[i % len(color_list)])

plt.show()

# 값을 두 집합 이상으로 분할
import numpy as np
import matplotlib.pyplot as plt

data = np.array([[5., 30., 45., 22.],
                 [5., 25., 50., 20.],
                 [1.,  2.,  1.,  1.]        
    ])

color_list = ['b', 'g', 'r']

X = np.arange(data.shape[1])

for i in range(data.shape[0]):
    plt.bar(X, data[i],
           bottom = np.sum(data[:i], axis=0),
           color = color_list[i % len(color_list)])

plt.show()
